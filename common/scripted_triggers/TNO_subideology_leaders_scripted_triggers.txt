# How this file works -
# Since you can't check subideology, we check by leader instead
# If you want to add a leader, simply add a new has_country_leader block to the proper ideology and you'll be all good
# Example of this:
#
# hasBolshevismLeader = {
# 	OR = {
# 		has_country_leader = { name = "Communist Guy 1" ruling_only = yes }
# 		has_country_leader = { name = "Communist Guy 2" ruling_only = yes }
# 		...
# 		has_country_leader = { name = "Communist Guy Infinity" ruling_only = yes }
# 	}
# }
#
#
# If your leader appears for two different subideologies that have different base ideologies, you can simply add then to both, as base ideology is checked first, before subideology
# Example of this:
#
# hasBolshevismLeader = {
# 	OR = {
# 		has_country_leader = { name = "Communist Guy" ruling_only = yes } # Only appears when Communist Guy is AuthSoc, as base ideology (AuthSoc) is checked before this
# 	}
# }
# 
# hasPopularFrontismLeader = {
# 	OR = {
# 		has_country_leader = { name = "Communist Guy" ruling_only = yes } # Only appears when Communist Guy is LibSoc, as base ideology (LibSoc) is checked before this
# 	}
# }
# 
# 
# If your leader appears for two different subideologies that have the same base ideology, you'll need to add an AND block around the has_country_leader, along with some sort of country flag to tell them apart
# Example of this:
#
# hasBolshevismLeader = {
# 	OR = {
# 		has_country_leader = { name = "Communist Guy 1" ruling_only = yes }
# 		has_country_leader = { name = "Communist Guy 2" ruling_only = yes }
#		AND = {
#			has_country_leader = { name = "Indecisive Communist Guy" ruling_only = yes }
#			TAG = { has_country_flag = Indecisive_Communist_Guy_Id1 }
#		}
# 	}
# }
# 
# hasStalinismLeader = {
# 	OR = {
# 		has_country_leader = { name = "Communist Guy 1" ruling_only = yes }
# 		has_country_leader = { name = "Communist Guy 2" ruling_only = yes }
#		AND = {
#			has_country_leader = { name = "Indecisive Communist Guy" ruling_only = yes }
#			TAG = { has_country_flag = Indecisive_Communist_Guy_Id2 }
#		}
# 	}
# }
#
