TNO_set_thermoelectric_plant_power = {
	#assume bonus_power is set as temp_var
	set_temp_variable = { current_power = tno_old_thermoelectric_plant_bonus_power }
	set_variable = { tno_old_thermoelectric_plant_bonus_power = bonus_power }
	subtract_from_temp_variable = { bonus_power = current_power }
	
	#reeee why doesn't this work
	#modify_building_resources = {
	#	building = thermoelectric_plant
	#	resource = power
	#	amount = var:bonus_power
	#}
	#Meta Effect to the rescue (curse me for suboptimization later)
	
	meta_effect = {
		text = {
			modify_building_resources = {
				building = thermoelectric_plant
				resource = power
				amount = [bonusPower]
			}
		}
		bonusPower = "[?bonus_power|.0]"
	}
}

TNO_update_safety_policy_effectiveness = {

	TNO_calculate_actual_safety_policy_effectiveness = yes

	set_variable = { tno_safety_policy_population_factor = tno_modifier_safety_policy_population_factor }
	set_variable = { tno_safety_policy_production_factory_max_efficiency_factor = tno_modifier_safety_policy_production_factory_max_efficiency_factor }
	set_variable = { tno_safety_policy_cost_per_capita = tno_modifier_safety_policy_cost_per_capita }
	set_variable = { tno_safety_policy_consumer_goods_production_factor = tno_modifier_safety_policy_consumer_goods_production_factor }

	set_temp_variable = { tno_safety_policy_positive_multiplier = tno_safety_effectiveness_actual }

	### In order for the GUI to work, policy effectiveness must be base-1 by default
	### This saves on memory
	multiply_temp_variable = { tno_safety_policy_positive_multiplier = 0.01 }

	### This looks insane, but it's important.
	### Policy Effectiveness followos a pattern of 1/2x+0.5, with x = policy effectiveness
	### This models that
	divide_temp_variable = { tno_safety_policy_positive_multiplier = 2 }
	add_to_temp_variable = { tno_safety_policy_positive_multiplier = 0.5 }
	set_temp_variable = { tno_safety_policy_negative_multiplier = 1 }
	subtract_from_temp_variable = { tno_safety_policy_negative_multiplier = tno_safety_policy_positive_multiplier }
	add_to_temp_variable = { tno_safety_policy_negative_multiplier = 1 }

	### Doing math to get the final answer - this is actually necessary, because variables tend to go negative as laws progress.
	### As a bonus, this automatically saves on calculations for variables that aren't in use.
	if = {
		limit = { check_variable = { tno_modifier_safety_policy_production_factory_max_efficiency_factor = 0 } }
		set_variable = { tno_safety_policy_production_factory_max_efficiency_factor = 0 }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_production_factory_max_efficiency_factor < 0 } }
		multiply_variable = { tno_safety_policy_production_factory_max_efficiency_factor = tno_safety_policy_negative_multiplier }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_production_factory_max_efficiency_factor > 0 } }
		multiply_variable = { tno_safety_policy_production_factory_max_efficiency_factor = tno_safety_policy_positive_multiplier }
	}
	

	# One of these blocks per modifier variable.
	if = {
		limit = { check_variable = { tno_modifier_safety_policy_population_factor = 0 } }
		set_variable = { tno_safety_policy_population_factor = 0 }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_population_factor < 0 } }
		multiply_variable = { tno_safety_policy_population_factor = tno_safety_policy_negative_multiplier }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_population_factor > 0 } }
		multiply_variable = { tno_safety_policy_population_factor = tno_safety_policy_positive_multiplier }
	}
	
	if = {
		limit = { check_variable = { tno_modifier_safety_policy_consumer_goods_production_factor = 0 } }
		set_variable = { tno_safety_policy_consumer_goods_production_factor = 0 }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_consumer_goods_production_factor < 0 } }
		multiply_variable = { tno_safety_policy_consumer_goods_production_factor = tno_safety_policy_positive_multiplier }
	}
	else_if = {
		limit = { check_variable = { tno_safety_policy_consumer_goods_production_factor > 0 } }
		multiply_variable = { tno_safety_policy_consumer_goods_production_factor = tno_safety_policy_negative_multiplier }
	}
