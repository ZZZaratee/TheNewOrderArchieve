			OR = {
				has_government = liberal_democracy

				has_government = conservative_democracy
			}
		}

			limit = {
				check_variable = { KOMLeftPower > KOMRightPower }
				NOT = {
					has_country_flag = KOM_leftist_coup

			country_event = {
				id = komicoup.1
				days = 3

KOM_get_most_likely_coupist_center = {
	if = {
		limit = {
			AND = {
				OR = {
					AND = {
						check_variable = { KOMStalinaInfluence > KOMVoznesenskyInfluence }
						check_variable = { KOMStalinaInfluence > KOMKosyginInfluence }
					}
					has_country_flag = KOM_democratic_coup
				}
				NOT = {
					OR = {
						has_country_flag = KOM_Stalina_Dead
						has_country_flag = KOM_Stalina_Arrested
						has_country_flag = KOM_Stalina_Retired
						has_country_flag = KOM_stalina_coup
					}
				}
			}
		}
		country_event = {
			id = komicoup.3
			days = 3
		}
	}
	else_if = {
		limit = {
			NOT = {
				OR = {
					AND = {
						OR = {
							has_country_flag = KOM_Voznesensky_Dead
							has_country_flag = KOM_Voznesensky_Arrested
							has_country_flag = KOM_Voznesensky_Retired
						}
						OR = {
							has_country_flag = KOM_Kosygin_Dead
							has_country_flag = KOM_Kosygin_Arrested
							has_country_flag = KOM_Kosygin_Retired
						}
					}
				}
				has_country_flag = KOM_democratic_coup
			}
		}
		country_event = {
			id = komicoup.4
			days = 3
		}
	}
	else = {
		KOM_generate_coup = yes
	}
}
